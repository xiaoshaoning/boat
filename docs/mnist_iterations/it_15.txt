- **资源预测显示**：每5个epoch显示一次资源使用预测，帮助用户合理规划计算资源。

```c
// 梯度统计数据结构
typedef struct {
    float total_norm;      // 梯度L2范数
    float max_grad;        // 最大梯度绝对值
    float min_grad;        // 最小梯度绝对值（非零）
    float mean_grad;       // 平均梯度绝对值
    float grad_clip_ratio; // 需要裁剪的梯度比例
    int param_count;       // 有梯度的参数数量
    int clipped_count;     // 需要裁剪的参数数量
} gradient_stats_t;

// 训练稳定性统计数据结构
typedef struct {
    float loss_mean;          // 近期损失平均值
    float loss_stddev;        // 损失标准差
    float loss_smoothness;    // 训练曲线平滑度
    float val_improvement;    // 验证损失改善率
    int patience_counter;     // 早期停止耐心计数器
    bool should_stop;         // 是否应停止训练
} stability_stats_t;

// 早期停止监控数据结构
typedef struct {
    int patience;           // 耐心值（epoch数）
    float min_delta;        // 最小改善阈值
    float best_val_loss;    // 最佳验证损失
    int epochs_no_improve;  // 无改善的epoch数
    bool stop_flag;         // 停止标志
} early_stopping_t;

// 资源预测数据结构
typedef struct {
    double avg_epoch_time;      // 平均epoch时间
    double time_remaining;      // 预计剩余时间
    double mem_growth_rate;     // 内存增长率
    double projected_mem_peak;  // 预计峰值内存
} resource_prediction_t;
```

### 梯度统计计算
```c
// 计算所有模型参数的梯度统计
gradient_stats_t compute_gradient_stats(mnist_model_t* model, float clip_threshold) {
    gradient_stats_t stats = {0};
    float sum_squares = 0.0f;
    float sum_abs = 0.0f;
    float max_val = 0.0f;
    float min_val = FLT_MAX;
    int total_params = 0;
    int clipped_params = 0;

    // 处理所有层的梯度张量（卷积层、全连接层）
    if (model->conv1) {
        PROCESS_GRADIENT_TENSOR(boat_conv_layer_get_grad_weight(model->conv1));
        PROCESS_GRADIENT_TENSOR(boat_conv_layer_get_grad_bias(model->conv1));
    }
    // ... 其他层

    stats.total_norm = sqrtf(sum_squares);
    stats.max_grad = max_val;
    stats.min_grad = (min_val == FLT_MAX) ? 0.0f : min_val;
    stats.mean_grad = total_params > 0 ? sum_abs / total_params : 0.0f;
    stats.grad_clip_ratio = total_params > 0 ? (float)clipped_params / total_params : 0.0f;

    return stats;
}
```

### 早期停止实现
```c
// 更新早期停止监控
void early_stopping_update(early_stopping_t* monitor, float val_loss, int epoch) {
    if (val_loss < monitor->best_val_loss - monitor->min_delta) {
        // 检测到改善
        monitor->best_val_loss = val_loss;
        monitor->epochs_no_improve = 0;
        printf("早期停止：最佳验证损失改善为%.4f（epoch %d）\n", val_loss, epoch + 1);
    } else {
        // 无改善
        monitor->epochs_no_improve++;
        if (monitor->epochs_no_improve >= monitor->patience) {
            monitor->stop_flag = true;
            printf("早期停止：耐心耗尽（%d个epoch无改善）\n", monitor->patience);
        }
    }
}
```

### 资源预测算法
```c
// 基于训练历史预测资源使用
resource_prediction_t predict_resources(const double* epoch_times, int current_epoch,
                                       int total_epochs, const boat_memory_stats_t* mem_stats,
                                       int window_size) {
    resource_prediction_t pred = {0};

    // 计算最近epoch的平均时间
    int start_epoch = current_epoch - window_size;
    if (start_epoch < 0) start_epoch = 0;
    int count = current_epoch - start_epoch + 1;

    double total_time = 0.0;
    for (int i = start_epoch; i <= current_epoch; i++) {
        total_time += epoch_times[i];
    }
    pred.avg_epoch_time = total_time / count;

    // 预测剩余时间
    int epochs_remaining = total_epochs - (current_epoch + 1);
    pred.time_remaining = pred.avg_epoch_time * epochs_remaining;

    // 预测内存使用（保守估计：当前峰值的110%）
    if (mem_stats && current_epoch >= 1) {
        double peak_mem = mem_stats->peak_allocated_bytes / (1024.0 * 1024.0);
        pred.projected_mem_peak = peak_mem * 1.1;
    }

    return pred;
}
```
