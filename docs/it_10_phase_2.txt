Implementation Options
----------------------

### Option 1: C++ Wrapper Approach (Recommended)
```cmake
# CMakeLists.txt updates
if(BOAT_WITH_PYTORCH)
    list(APPEND BOAT_SOURCES src/format/pytorch.cpp)  # Change to .cpp
    set(Torch_DIR "${CMAKE_SOURCE_DIR}/external/libtorch-shared-with-deps-latest/libtorch/share/cmake/Torch")
    find_package(Torch REQUIRED)
    target_link_libraries(boat PUBLIC Torch::Torch)
endif()
```

**Advantages**:
- Direct access to LibTorch C++ API, full functionality
- Maintains existing C interface for compatibility
- Simple CMake configuration
- Optimal performance with minimal data copying

**Disadvantages**:
- Requires C++ compilation for part of the project
- Need to handle C++ exceptions in C code

### Option 2: Pure C Interface
- Keep existing `pytorch.c` as C file
- Create separate C++ module that exports C-compatible functions
- Use `extern "C"` wrappers for all LibTorch interactions

**Advantages**:
- Clear separation between C and C++ code
- No C++ compilation in main library

**Disadvantages**:
- More complex build configuration
- Additional function call overhead
- Duplicate code for interface wrapping

### Option 3: Hybrid Compilation
```cmake
# Separate C++ module
add_library(boat_pytorch STATIC src/format/pytorch.cpp)
target_link_libraries(boat_pytorch PUBLIC Torch::Torch)
target_link_libraries(boat PRIVATE boat_pytorch)
```

**Advantages**:
- Isolates C++ dependencies
- Can be conditionally linked

**Disadvantages**:
- More complex project structure
- Additional linking requirements

### Step 2: Implement Real PyTorch Loader (C++)
```cpp
// pytorch.cpp - LibTorch C++ API implementation
#include <torch/script.h>
#include <torch/torch.h>
#include <vector>
#include <string>
#include <exception>

#include "boat/format/pytorch.h"
#include "boat/model.h"
#include "boat/tensor.h"
#include "boat/memory.h"

// Data type mapping helper
static boat_dtype_t torch_dtype_to_boat(torch::Dtype dtype) {
    switch (dtype) {
        case torch::kFloat32: return BOAT_DTYPE_FLOAT32;
        case torch::kFloat64: return BOAT_DTYPE_FLOAT64;
        case torch::kFloat16: return BOAT_DTYPE_FLOAT16;
        case torch::kInt32:   return BOAT_DTYPE_INT32;
        case torch::kInt64:   return BOAT_DTYPE_INT64;
        case torch::kInt8:    return BOAT_DTYPE_INT8;
        case torch::kUInt8:   return BOAT_DTYPE_UINT8;
        case torch::kBool:    return BOAT_DTYPE_BOOL;
        default:              return BOAT_DTYPE_FLOAT32;
    }
}

// Shape conversion helper
static void torch_shape_to_boat(const torch::Tensor& tensor,
                                int64_t** shape_ptr,
                                size_t* ndim_ptr) {
    auto sizes = tensor.sizes();
    *ndim_ptr = sizes.size();
    *shape_ptr = (int64_t*)boat_malloc(sizeof(int64_t) * (*ndim_ptr), BOAT_DEVICE_CPU);

    for (size_t i = 0; i < *ndim_ptr; i++) {
        (*shape_ptr)[i] = sizes[i];
    }
}

extern "C" {
    boat_model_t* boat_pytorch_load(const char* filename) {
        try {
            // Load PyTorch model using LibTorch
            torch::jit::script::Module module;
            try {
                module = torch::jit::load(filename);
            } catch (const c10::Error& e) {
                fprintf(stderr, "Failed to load PyTorch model %s: %s\n",
                        filename, e.what());
                return NULL;
            }

            // Create Boat model
            boat_model_t* model = boat_model_create();
            if (!model) {
                fprintf(stderr, "Failed to create Boat model\n");
                return NULL;
            }

            // Extract model parameters
            auto params = module.named_parameters();
            for (const auto& param : params) {
                torch::Tensor torch_tensor = param.value();

                // Skip non-contiguous tensors (need to make contiguous first)
                if (!torch_tensor.is_contiguous()) {
                    torch_tensor = torch_tensor.contiguous();
                }

                // Convert shape
                int64_t* shape = NULL;
                size_t ndim = 0;
                torch_shape_to_boat(torch_tensor, &shape, &ndim);

                // Convert data type
                boat_dtype_t dtype = torch_dtype_to_boat(torch_tensor.dtype());

                // Get tensor data
                void* data = torch_tensor.data_ptr();
                size_t nbytes = torch_tensor.nbytes();

                // Create Boat tensor
                boat_tensor_t* boat_tensor = boat_tensor_create(shape, ndim, dtype, BOAT_DEVICE_CPU);
                if (!boat_tensor) {
                    boat_free(shape);
                    boat_model_free(model);
                    return NULL;
                }

                // Copy data (could be optimized to zero-copy in some cases)
                void* boat_data = boat_tensor_data(boat_tensor);
                memcpy(boat_data, data, nbytes);

                // TODO: Add tensor to model with appropriate name mapping
                // For now, store in model user data or parameter list

                // Clean up
                boat_free(shape);
                boat_tensor_free(boat_tensor);
            }

            // Extract model buffers (non-parameter tensors)
            auto buffers = module.named_buffers();
            for (const auto& buffer : buffers) {
                // Similar processing as parameters
                // These might include running mean/variance for batch norm, etc.
            }

            // TODO: Extract model architecture/graph structure
            // This is more complex and requires parsing the TorchScript graph

            return model;

        } catch (const std::exception& e) {
            fprintf(stderr, "PyTorch loading error: %s\n", e.what());
            return NULL;
        }
    }

    // Other PyTorch API functions with similar implementation...
}
```

### Step 5: Error Handling Framework
```cpp
// Unified error handling
class PyTorchLoaderError : public std::exception {
private:
    std::string message_;
    boat_error_t boat_error_;

public:
    PyTorchLoaderError(const std::string& msg, boat_error_t err = BOAT_ERROR_UNKNOWN)
        : message_(msg), boat_error_(err) {}

    const char* what() const noexcept override { return message_.c_str(); }
    boat_error_t boat_error() const { return boat_error_; }
};

// Error translation function
static boat_error_t translate_exception(const std::exception& e) {
    if (dynamic_cast<const c10::Error*>(&e)) {
        return BOAT_ERROR_FORMAT;
    } else if (dynamic_cast<const std::bad_alloc*>(&e)) {
        return BOAT_ERROR_OUT_OF_MEMORY;
    } else {
        return BOAT_ERROR_UNKNOWN;
    }
}
```

Risk Assessment and Mitigation
------------------------------

### High Risk Areas
1. **C/C++ Interoperability**: Mixed language compilation issues
   - *Mitigation*: Use well-defined C interfaces, thorough testing

2. **LibTorch API Stability**: Version compatibility issues
   - *Mitigation*: Pin to specific LibTorch version, API compatibility layer

3. **Memory Management**: Leaks between different allocators
   - *Mitigation*: Rigorous memory testing, valgrind/asan checks

4. **Performance Overhead**: Excessive data copying
   - *Mitigation*: Profile and optimize hot paths, consider zero-copy approaches

### Medium Risk Areas
1. **Model Graph Conversion**: Complex TorchScript graph parsing
   - *Mitigation*: Start with simple models, gradual complexity increase

2. **Error Propagation**: Exceptions crossing language boundaries
   - *Mitigation*: Comprehensive error translation and testing

### Low Risk Areas
1. **Data Type Mapping**: Straightforward conversion table
2. **Basic Parameter Loading**: Simple tensor data extraction
