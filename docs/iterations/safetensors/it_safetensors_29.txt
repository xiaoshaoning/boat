#### 2. Adam 优化器算法实现
```c
// Adam 优化器状态结构
typedef struct {
    float learning_rate;
    float beta1;
    float beta2;
    float epsilon;
    int64_t timestep;

    // 一阶矩估计（均值）
    boat_tensor_t** m;
    // 二阶矩估计（未中心化方差）
    boat_tensor_t** v;
    // 参数指针数组
    boat_tensor_t** params;
    // 梯度指针数组
    boat_tensor_t** grads;
    size_t num_params;
} boat_adam_state_t;

// Adam 参数更新核心算法
static void adam_update_param(float* param, float* grad, float* m, float* v,
                              float lr, float beta1, float beta2,
                              float beta1_pow_t, float beta2_pow_t,
                              float epsilon, int64_t t) {
    // 更新一阶矩估计
    *m = beta1 * (*m) + (1.0f - beta1) * (*grad);

    // 更新二阶矩估计
    *v = beta2 * (*v) + (1.0f - beta2) * (*grad) * (*grad);

    // 计算偏置校正后的矩估计
    float m_hat = *m / (1.0f - beta1_pow_t);
    float v_hat = *v / (1.0f - beta2_pow_t);

    // 应用参数更新
    *param -= lr * m_hat / (sqrtf(v_hat) + epsilon);
}
```
